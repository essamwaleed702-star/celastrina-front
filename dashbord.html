🗄️ نموذج Products Table في SQLite

-- في QKFASTAPI بتاعك، اعمل الجدول ده:

CREATE TABLE products (
    id        INTEGER PRIMARY KEY AUTOINCREMENT,
    name      TEXT    NOT NULL,
    category  TEXT    NOT NULL,  -- 'bag' أو 'trousse'
    price     REAL    NOT NULL,
    img       TEXT    NOT NULL,  -- URL أو path للصورة
    gallery   TEXT,              -- JSON array كـ string
    desc      TEXT,
    fomo      TEXT,              -- رسالة FOMO
    care      TEXT,              -- تعليمات العناية
    tags      TEXT,              -- JSON array كـ string
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
